#version 400

in vec3 textureCoords;
out vec4 out_Color;

uniform samplerCube cubeMap1;
uniform samplerCube cubeMap2;
uniform vec3 FogColor;
uniform float blendFactor;

const float lowerLimit = 0;
const float upperLimit = 30;


void main(void){

    float FogFactor = (textureCoords.y - lowerLimit)/(upperLimit - lowerLimit);
    FogFactor = clamp(FogFactor,0.0,1.0);
    vec4 Color1 = texture(cubeMap1, textureCoords);
    vec4 Color2 = texture(cubeMap2,textureCoords);
    vec4 Color = mix(Color1,Color2,blendFactor);
    out_Color = mix(vec4(FogColor,1.0),Color,FogFactor);
}
